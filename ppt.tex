\begin{document}

\title{A Simple Presentation}
\author{A \LaTeX{}er}

\begin{frame}{main page}
  \titlepage
\end{frame}

\begin{frame}{Hello, world!}
  This is a "Hello, world!" example of writing presentation with \LaTeX{} beamer.
\end{frame}
%一套幻灯片是由若干幅画面（frame）组成的，每个 frame 可以有一个标题（frametitle）
% 若为example则不能控制标题，若为exampleblock则可以控制标题
\begin{frame}\frametitle{Examples}
  \begin{example}{sim example}
    Itemize example:
    \begin{itemize}
    \item Hello, world! is simple;
    \item \LaTeX{} beamer is easy.
    \end{itemize}
  \end{example}
  \begin{exampleblock}{Enumerate example:}
    \begin{enumerate}
    \item \LaTeX{} beamer is powerful, and
    \item beautiful.
    \end{enumerate}
  \end{exampleblock}
\end{frame}

%逐列显示
